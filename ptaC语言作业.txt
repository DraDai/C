#include <stdio.h>//计算华氏度
int main()
{
	double F = 100;
	int C;
	C = 5 * (F - 32) / 9;
	printf("fahr = 100, celsius = %d",C);
	return 0;
}

#include <stdio.h>//计算居民水费
#include<math.h>
int main()
{
	double x, y;
	scanf("%lf", &x);
	if (x <= 15)
		y = 4 * x / 3;
	else
		y = 2.5 * x - 10.5;
	printf("f(%.2lf) = %.2lf", x, y);
	return 0;
}
