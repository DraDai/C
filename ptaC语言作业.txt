		printf("Invalid.");
	else
	{
		printf("fahr celsius\n");

#include<stdio.h>//计算1到n的和
int main()
{

	for (i = 2, p = 1, s = -1.0;i <= n;i++)

	return 0;
}

#include<stdio.h>//华氏温度转换表
#include<math.h>
int main()
{
	int m, n;
	double p;
	scanf("%d%d", &m, &n);
	p = m*m + 1.0 / m;
	for (;m < n;m++)
	{
		p += pow(m + 1.0, 2) + 1.0 / (m + 1);
	}
	printf("sum = %.6lf", p);
	return 0;
}
