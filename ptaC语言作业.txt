#include <stdio.h>//计算华氏度

{

}

#include <stdio.h>//计算居民水费
#include<math.h>
int main()
{
	double x, y;
	scanf("%lf", &x);
	if (x <= 15)
		y = 4 * x / 3;
	else
		y = 2.5 * x - 10.5;
	printf("f(%.2lf) = %.2lf", x, y);
	return 0;
}





#include<stdio.h>//计算华氏温度
int main()
{
	int a = 152, x, y, z;
	x = a / 100;//百位数字
	y = a % 100 / 10;//十位数字
	z = a % 10;//个位数字
	printf("152 = %d + %d*10 + %d*100", z, y, x);
	return 0;
}
