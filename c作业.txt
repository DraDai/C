int main()

int main()
{
	double r, c, s;
	scanf("%lf", &r);
	c = 3.14 * 2 * r;
	s = 3.14 * (r*r);
	printf("%lf %lf", c, s);
	return 0;
}



#include <stdio.h>//比较三个整数大小
int main()
{
	int a, b, c;
	scanf("%d%d%d", &a, &b, &c);
	if (a > b)
		if (a > c)
			printf("%d", a);
		else
			printf("%d", c);
	else
		if (b > c)
			printf("%d", b);
		else
			printf("%d", c);
	return 0;
}
